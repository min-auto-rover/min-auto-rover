62Vision::init_dictionary(void) {
63 int markerSize = 4;
64 int nMarkers = 2;
65 cv::Mat bytesList(nMarkers, markerSize * markerSize, CV_8UC1);
66 log_info << "creating dictionary";
67 dictionary = cv::makePtr<cv::aruco::Dictionary>();
68 dictionary->markerSize = 4;
69 dictionary->maxCorrectionBits = 3;
70 cv::Mat marker0 = (cv::Mat_<uchar>(4, 4) <<
71 0, 1, 0, 1,
72 0, 0, 1, 1,
73 0, 0, 1, 0,
74 1, 1, 0, 0
75 );
76 cv::Mat marker1 = (cv::Mat_<uchar>(4, 4) <<
77 1, 0, 1, 0,
78 1, 1, 1, 1,
79 1, 0, 0, 0,
80 1, 1, 1, 1
81 );
82 dictionary->bytesList.push_back(
83 cv::aruco::Dictionary::getByteListFromBits(marker0));
84 dictionary->bytesList.push_back(
85 cv::aruco::Dictionary::getByteListFromBits(marker1));

99void
100Vision::vision_main_loop(void) {
101 cv::VideoCapture cap(gst_pipeline, cv::CAP_GSTREAMER);
102 cv::Mat frame;
103
104 for (;;) {
105 cap.read(frame);
106 if (frame.empty()) {
107 log_error << "Empty frame captured!";
108 continue;
109 }
110 std::vector<std::vector<cv::Point2f> > corners;
111 std::vector<int> ids;
112 cv::aruco::detectMarkers(frame, dictionary, corners, ids);
113 std::vector<Marker> markers;
114 if (corners.size()) {
115
116 for (size_t i = 0; i < corners.size(); i++) {
117 /*std::cout << "i=" << i << " id=" << ids[i]
118 << " corners0=" << corners[i][0]
119 << " corners1=" << corners[i][1]
120 << " corners2=" << corners[i][2]
121 << " corners3=" << corners[i][3]
122 << std::endl;*/
123 Marker this_marker;
124 this_marker.id = ids[i];
125 this_marker.corner0 = corners[i][0];
126 this_marker.corner1 = corners[i][1];
127 this_marker.corner2 = corners[i][2];
128 this_marker.corner3 = corners[i][3];
129 this_marker.set_marker();
130 markers.push_back(this_marker); // FIXME abi change
131 }
132 }
133 if (!markers.empty()) {
134 std::sort(markers.begin(), markers.end(),
135 [](const Marker &a, const Marker &b) {
136 return a.area > b.area;
137 }); //FIXME abi change
138 // If the largest marker is start or goal, do the tasks
139 // accordingly.
140 if (markers.front().id == START_MARKER_ID) {
141 // TODO print *
142 Uart::send('*');
143 continue;
144 }
145 if (markers.front().id == GOAL_MARKER_ID) {
146 // TODO print @
147 Uart::send('@');
148 continue;
149 }
150 std::vector<Marker> left_markers, right_markers;
151 for (const auto& m : markers) {
152 switch (m.id) {
153 case GATE_MARKER_LEFT:
154 left_markers.push_back(m); //FIXME abi change
155 break;
156 case GATE_MARKER_RIGHT:
157 right_markers.push_back(m); //FIXME abi change
158 break;
159 default:
160 log_error << "Found a marker without "
161 "a valid marker id";
162 }
163 }
164 if (left_markers.empty() || right_markers.empty()) {
165 // TODO print ?
166 Uart::send('?');
167 log_warn << "Tags not enough to form pairs";
168 log_debug << "... left markers="
169 + std::to_string(left_markers.size());
170 log_debug << "... right markers="
171 + std::to_string(right_markers.size());
172 continue;
173 }
174
175 /* Pair markers into gate */
176
177 Marker curr_left_marker, curr_right_marker;
178#if MATCH_GATE_PAIR_ALGO == largest_mix_and_match
179 // If the largest left and right markers have similar
180 // area, then pair them;
181 // Else if the largest left (right) matches the second
182 // largest right (left), pair them;
183 // Else, fallback, just match the largest ones.
184 if (left_markers.front().centre.x
185 < right_markers.front().centre.x
186 && std::abs(left_markers.front().area -
187 right_markers.front().area)
188 <= GATE_PAIR_D_AREA_THRESH) {
189 curr_left_marker = left_markers.front();
190 curr_right_marker = right_markers.front();
191 } else if (left_markers.size() > 1
192 && left_markers.at(1).centre.x
193 < right_markers.front().centre.x
194 && std::abs(left_markers.at(1).area -
195 right_markers.front().area)
196 <= GATE_PAIR_D_AREA_THRESH) {
197 curr_left_marker = left_markers.at(1);
198 curr_right_marker = right_markers.front();
199 } else if (right_markers.size() > 1
200 && std::abs(left_markers.front().area -
201 right_markers.at(1).area)
202 <= GATE_PAIR_D_AREA_THRESH) {
203 curr_left_marker = left_markers.front();
204 curr_right_marker = right_markers.at(1);
205 } else {
206 log_info << "Match gate pair algorithm "
207 "largest_mix_and_match fallback";
208 curr_left_marker = left_markers.front();
209 curr_right_marker = right_markers.front();
210 }
211#elif MATCH_GATE_PAIR_ALGO == forall_left_try_right
212 bool pair_found = false;
213 for (auto l : left_markers) {
214 for (auto r : right_markers) {
215 if (l.centre.x < r.centre.x
216 && std::abs(l.area - r.area)
217 <= GATE_PAIR_D_AREA_THRESH) {
218 curr_left_marker = l;
219 curr_right_marker = r;
220 pair_found = true;
221 }
222 }
223 }
224 if (!pair_found) {
225 log_info << "Match gate pair algorithm "
226 "forall_left_try_right fallback";
227 curr_left_marker = left_markers.front();
228 curr_right_marker = right_markers.front();
229 }
230#else
231# error "MATCH_GATE_PAIR_ALGO is undefined or erroneous"
232#endif
233 double gate_x = (curr_left_marker.centre.x +
234 curr_right_marker.centre.x) / 2.0;
235 char gate_x_char = (gate_x / FRAME_WIDTH
236 * 26.0) + 'A';
237 if (gate_x_char > 'Z')
238 gate_x_char = 'Z';
239
240 /* Switch letter case */
241
242 double this_gate_width = curr_right_marker.centre.x
243 - curr_left_marker.centre.x;
244#if PASS_GATE_ALGO == marker_area
245 if (last_left_marker_area - curr_left_marker.area
246 > PROCEED_D_AREA_THRESH
247 && last_right_marker_area - curr_right_marker.area
248 > PROCEED_D_AREA_THRESH) {
249#elif PASS_GATE_ALGO == gate_width
250 if (last_gate_width - this_gate_width
251 > PROCEED_D_GATE_WIDTH_THRESH) {
252#else
253# error "PASS_GATE_ALGO is undefined or erroneous"
254 if (0) {
255#endif
256 // note that the code here is in if body
257 // which means gate passed
258 gate_passed++;
259 } // end if (see preprocessors)
260 last_gate_width = this_gate_width;
261 last_left_marker_area = curr_left_marker.area;
262 last_right_marker_area = curr_right_marker.area;
263
264 char gate_output_char = (gate_passed % 2 == 0) ?
265 gate_x_char : std::tolower(gate_x_char);
266
267 log_debug << "G-char="
268 + std::to_string(gate_output_char)
269 + ", G-passed="
270 + std::to_string(gate_passed)
271 + ", G-x="
272 + std::to_string(gate_x)
273 + ", G-w="
274 + std::to_string(this_gate_width)
275 + ", F-w="
276 + std::to_string(FRAME_WIDTH);
277 // TODO print gate_output_char
278 Uart::send(gate_output_char);
279 } else {
280 log_info << "No markers seen";
281 Uart::send('?');
282 }
283 }
