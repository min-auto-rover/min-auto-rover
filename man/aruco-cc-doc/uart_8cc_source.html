37int
38Uart::init_uart(void) {
39 int fd = -1;
40#ifdef ENABLE_OUTPUT
41 fd = open(UART_PORT, O_RDWR | O_NOCTTY | O_NDELAY); // FIXME RdWr or Wonly
42 if (fd < 0) {
43 log_crit << "Could not open UART, see below for errno";
44 log_crit << std::to_string(errno);
45 log_warn << "Errno printed, Quitting now";
46 exit(EXIT_FAILURE);
47 }
48 struct termios options;
49 tcgetattr(fd, &options);
50 termios_setup(options); //macro
51 cfsetspeed(&options, B9600);
52 tcflush(uart_fd, TCIFLUSH);
53 if (tcsetattr(fd, TCSANOW, &options) != 0) {
54 log_crit << "Could not setup serial, see below for errno";
55 log_crit << std::to_string(errno);
56 log_warn << "Errno printed, Quitting now";
57 exit(EXIT_FAILURE);
58 }
59#endif
60#ifdef OUTPUT_TO_STDOUT
61 fd = open("/dev/stdout", O_WRONLY);
62#endif
63 file = fd;
64 initialised = true;
65 log_info << "output init done";
66 return fd;
67}
