31Logger::Logger(const std::string& filename, LogLevel level)

56Logger::print_log_level(LogLevel loglevel) const {

72void
73Logger::log(const std::string& message, LogLevel level) {
74 if (logfile && level >= loglevel) {
75 std::time_t now = std::time(nullptr);
76 char time_str[200];
77 std::strftime(time_str, sizeof time_str, "%Y-%m-%d %H:%M:%S",
78 std::localtime(&now));
79 fprintf(logfile, "%s [%s] \t%s\n", time_str,
80 print_log_level(level).c_str(), message.c_str());
81 fflush(logfile);
82#ifdef LOG_COPY_STDERR
83 fprintf(stderr, "%s [%s] \t%s\n", time_str,
84 print_log_level(level).c_str(), message.c_str());
85#endif
86 }
87}
